project( ecml CXX )

include( ecbuild_system NO_POLICY_SCOPE )

ecbuild_requires_macro_version( 1.9 )

                     HEADER_DESTINATION ${INSTALL_INCLUDE_DIR}
                     SOURCES            ${ecml_srcs}
